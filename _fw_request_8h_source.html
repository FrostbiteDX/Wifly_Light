31 REF.addr[3] = ((ADDRESS) & 0xff000000) >> 24; \
32 REF.addr[2] = ((ADDRESS) & 0x00ff0000) >> 16; \
33 REF.addr[1] = ((ADDRESS) & 0x0000ff00) >> 8; \
34 REF.addr[0] = ((ADDRESS) & 0x000000ff); \
35 REF.red = ((ARGB) & 0x00ff0000) >> 16; \
